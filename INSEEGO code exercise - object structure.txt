public class Order
{
  public var OrderId;
  public const Company Co;
    
  public Order(Company newCo) {
    // Generate new unique ID
    OrderId = Guid.NewGuid();
    Co = newCo;
  }

  public string toString(){
    return
      "OrderId: " + OrderId.toString() + ", " +
      "Product: " + Co.getProduct() + ", " +
      "Plan: " + Co.getPlan();
  }
}
